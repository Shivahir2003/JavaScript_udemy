        var a=Math.ceil(2.55);
        document.write(a); // 3


        a=Math.floor(2.55);
        document.write(a); // 2


        a=Math.round(2.55);
        document.write(a); //3


        a=Math.trunc(2.55); 
        document.write(a); // 2


        a=Math.max(2,50 ,5 ,100 ,54);
        document.write(a); //100


        a=Math.min(2,50 ,5 ,100 ,54);
        document.write(a);  // 2


        a=Math.sqrt(125);
        document.write(a); // 25


        a=Math.cbrt(8);
        document.write(a); //2


        a=Math.pow(2, 4);
        document.write(a);  //16


        a=Math.random();
        document.write(a); // Random number


        a=Math.abs(5.345345);
        document.write(a); //5.345345


        a=Math.PI;
        document.write(a); //3.14
